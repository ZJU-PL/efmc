[docs]def parse_chc(filename: str, to_real_type: bool):
    """Parse CHC file"""
    with open(filename, "r", encoding="utf-8") as f:
        res = f.read()
        ss = CHCParser(res, to_real_type)
        return ss.get_transition_system()
